
### 例子00

说明
1. 指出整形、整形指针、整形指针的值关于sizeof的不同

```
➜  ~ cat a.cpp
#include <cstdio>

int main() {
	int a1 = 1;
	int * a2 = &a1;
	printf("%lu\n", sizeof(a1));
	printf("%lu\n", sizeof(a2));
	printf("%lu\n", sizeof(*a2));

	char b1 = '1';
	char * b2 = &b1;
	printf("%lu\n", sizeof(b1));
        printf("%lu\n", sizeof(b2));
        printf("%lu\n", sizeof(*b2));

	unsigned long long c1 = 111111;
	unsigned long long * c2 = &c1;
        printf("%lu\n", sizeof(c1));
        printf("%lu\n", sizeof(c2));
        printf("%lu\n", sizeof(*c2));
	return 0;
}

```

### 例子01

说明
1. 可以使用“结构体指针赋值给整形指针，整形指针再复制给结构体指针”的方法，以记录指针位置

```
➜  ~ cat a.cpp
#include <cstdio>

struct A {
	int a;
	char b;
	long long c;
};

int main() {
	A a = {1, '2', 3LL};
	printf("%d %c %lld\n", a.a, a.b, a.c);

	A * b = &a;
	printf("%p\n", b);
	printf("%d %c %lld\n", b->a, b->b, b->c);

	unsigned char * c = (unsigned char *) b;
	printf("%p\n", c);

	A * d = (A *) c;
	printf("%p\n", d);
	printf("%d %c %lld\n", d->a, d->b, d->c);

	return 0;
}
➜  ~ g++ a.cpp -o exec_a; ./exec_a
1 2 3
0x7ffeedea06c8
1 2 3
0x7ffeedea06c8
0x7ffeedea06c8
1 2 3
```
